function updateDisplay(reset = true) { const q = el('searchInput').value.toLowerCase().trim(); const posFilter = el('filterPos').value.toLowerCase().trim(); const areaFilter = el('filterArea').value.toLowerCase().trim(); filteredData = DATABASE_CONTENT.filter(item => { // Подготовка данных из строки (с защитой от пустых значений) const g = (item.column_A_greek || []).join(' ').toLowerCase(); const r = (item.column_B_russian || []).join(' ').toLowerCase(); const itemPos = (item.column_D_part_of_speech || "").toLowerCase().trim(); const itemArea = (item.column_E_area || "").toLowerCase().trim(); // Логика фильтрации const matchesSearch = !q || g.includes(q) || r.includes(q); // Теперь проверяем, содержится ли выбранный фильтр в строке const matchesPos = !posFilter || itemPos.includes(posFilter); const matchesArea = !areaFilter || itemArea.includes(areaFilter); return matchesSearch && matchesPos && matchesArea; }); if (reset) { el('tableBody').innerHTML = ''; renderedCount = 0; renderMore(); } }
